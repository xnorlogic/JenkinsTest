pipeline {
    agent {
        node {                          
            label 'Zorin_Node'
        }
    }
	environment {
		SRC_DIR = 'src'
		APP_NAME = 'Test_GCC'
	}
    stages {
		stage('Stage 1') {
            steps {
                sh 'echo hello'
            }
		}
		stage('Make the application') {
            steps {
				sh 'cd application'
				sh 'make'
            }
		}
		stage('Run the binary') {
            steps {
				sh './application/bin/${APP_NAME}.deb'
            }
		}
    }
	post {
		always {
            sh 'echo One way or another, I have finished'
        }
        success {
            sh 'echo I succeeeded!'	
        }
        unstable {
            sh 'echo I am unstable :/'
        }
        failure {
            sh 'echo I failed :('
        }
        changed {
            sh 'echo Things were different before...'
        }
	}
}
